# Custom Categorical Focal Loss with per-class alpha
def categorical_focal_loss(gamma=4.0, alpha=[0.01, 0.98, 0.01]):  # Adjusted alpha for stronger SVEB emphasis
    def focal_loss(y_true, y_pred):
        y_pred = tf.clip_by_value(y_pred, tf.keras.backend.epsilon(), 1. - tf.keras.backend.epsilon())
        cross_entropy = -y_true * tf.math.log(y_pred)
        alpha_t = tf.constant(alpha, dtype=tf.float32)
        weight = alpha_t * y_true * tf.pow((1 - y_pred), gamma)
        loss = weight * cross_entropy
        loss = tf.reduce_sum(loss, axis=1)
        return tf.reduce_mean(loss)
    return focal_loss

def load_incart_data(record_ids=range(1, 76)):
    X_raw, y, groups = [], [], []
    for rec_id in record_ids:
        rec_name = f'I{rec_id:02d}'
        try:
            record = wfdb.rdrecord(f'incartdb/{rec_name}', channels=[1, 4, 10])  # Lead II, V1, V5
            ann = wfdb.rdann(f'incartdb/{rec_name}', 'atr')

            for i, symbol in enumerate(ann.symbol):
                if symbol not in sum(AAMI_MAPPING.values(), []): continue
                label = next(k for k, v in AAMI_MAPPING.items() if symbol in v)

                peak = ann.sample[i]
                start = max(0, peak - 128)
                end = min(len(record.p_signal), peak + 128)
                if end - start < 256: continue

                signal = record.p_signal[start:end]  # Shape: (256, 3)
                signal = (signal - signal.mean(axis=0)) / (signal.std(axis=0) + 1e-8)  # Normalize per lead

                X_raw.append(signal)
                y.append(label)
                groups.append(record_to_patient[rec_name])
        except Exception as e:
            print(f"Error loading {rec_name}: {e}")

    X_raw = np.array(X_raw)  # Shape: (n_beats, 256, 3)
    y = np.array(y)
    groups = np.array(groups)

    print(f"Loaded {len(X_raw)} beats. Label dist: {Counter(y)}")
    return X_raw, y, groups

def augment_data(X, y, seed=42):
    np.random.seed(seed)
    from scipy.signal import resample
    X_aug = X.copy()
    # Gaussian noise
    noise = np.random.normal(0, 0.05, X.shape)
    X_aug += noise
    # Gaussian smoothing (σ=0.5)
    for i in range(len(X_aug)):
        for lead in range(X_aug.shape[2]):
            X_aug[i, :, lead] = gaussian_filter1d(X_aug[i, :, lead], sigma=0.5)
    # Time warping (±5% for SVEB, ±2% for others)
    for i in range(len(X_aug)):
        scale = np.random.uniform(0.95, 1.05) if y[i] == sveb_code else np.random.uniform(0.98, 1.02)  # ±5% for SVEB
        new_length = int(256 * scale)
        for lead in range(X_aug.shape[2]):
            resampled = resample(X_aug[i, :, lead], new_length)
            if len(resampled) < 256:
                resampled = np.pad(resampled, (0, 256 - len(resampled)), mode='edge')
            X_aug[i, :, lead] = resampled[:256]
    # Time shift (<10 samples)
    shifts = np.random.randint(-5, 5, len(X))
    for i in range(len(X)):
        X_aug[i] = np.roll(X_aug[i], shifts[i], axis=0)
    return X_aug, y

# CBAM Layer (Channel + Spatial Attention)
class CBAMLayer(Layer):
    def __init__(self, ratio=8, kernel_size=7, **kwargs):
        super(CBAMLayer, self).__init__(**kwargs)
        self.ratio = ratio
        self.kernel_size = kernel_size

    def build(self, input_shape):
        self.channels = input_shape[-1]
        self.shared_dense1 = Dense(self.channels // self.ratio, activation='relu', kernel_initializer='he_normal', use_bias=True, bias_initializer='zeros')
        self.shared_dense2 = Dense(self.channels, activation=None, kernel_initializer='he_normal', use_bias=True, bias_initializer='zeros')
        self.spatial_conv = Conv1D(1, kernel_size=self.kernel_size, padding='same', activation='sigmoid', kernel_initializer='he_normal', use_bias=False)
        super(CBAMLayer, self).build(input_shape)

    def call(self, x):
        # Channel attention
        avg_pool = tf.reduce_mean(x, axis=1)
        max_pool = tf.reduce_max(x, axis=1)
        avg_out = self.shared_dense2(self.shared_dense1(avg_pool))
        max_out = self.shared_dense2(self.shared_dense1(max_pool))
        channel_att = tf.nn.sigmoid(avg_out + max_out)
        x = x * tf.expand_dims(channel_att, axis=1)

        # Spatial attention
        avg_spatial = tf.reduce_mean(x, axis=2, keepdims=True)
        max_spatial = tf.reduce_max(x, axis=2, keepdims=True)
        spatial_concat = tf.concat([avg_spatial, max_spatial], axis=2)
        spatial_att = self.spatial_conv(spatial_concat)
        x = x * spatial_att
        return x

def build_cnn(input_shape=(256, 3), filters_1=64, filters_2=128, filters_3=256, filters_4=256, filters_5=512, kernel_size=5, lr=2e-4):

    x = Add()([x, Conv1D(filters_2, 1, padding='same')(residual)])
